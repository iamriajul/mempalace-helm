Every client connection must include:
  Authorization: Bearer <your-token>

Retrieve the token:
  kubectl get secret {{ include "mempalace.fullname" . }}-auth \
    -n {{ .Release.Namespace }} \
    -o jsonpath='{.data.token}' | base64 -d && echo

  claude mcp add mempalace --transport http \
    {{- if .Values.auth.enabled }}

    {{- end }}
